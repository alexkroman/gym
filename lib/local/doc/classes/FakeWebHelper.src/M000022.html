  def http_server_errors(urls = [], &block)
    {
      500 => 'Internal Server Error',
      501 => 'Not Implemented',
      502 => 'Bad Gateway',
      503 => 'Service Unavailable',
      504 => 'Gateway Timeout',
      505 => 'HTTP Version Not Supported'
    }.each do |code, message|
      fake_web(urls, :status => [code, message], &block)
    end
  end
