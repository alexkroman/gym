  def stop_fake_web(*args)
    if args.empty?
      FakeWeb.clean_registry
    else
      args.each do |url|
        # Delete each URI from the registry.
        uri = URI.parse(url.to_s)
        uri.path = '/' if uri.path.blank?
        FakeWeb::Registry.instance.uri_map.delete(uri)
      end
    end
  end
