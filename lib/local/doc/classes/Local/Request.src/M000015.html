  def path(query)
      string  =   "http://local.yahooapis.com/LocalSearchService/V2/localSearch?"
      string  +=  "appid=#{Local.app_id}&query=#{CGI::escape(query)}"
      @options.each do |key, value|
        string += "&#{key.to_s}=#{CGI::escape(value.to_s)}"
      end
      string
  end
